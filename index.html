Javascript Vanilla

Code

        alert("Ready!");

Double Clicked

Explanation: When an HTML item has been double clicked

Javascript Vanilla

Code 
    document.getElementById("demo").ondblclick = function() {myFunction()};

    function myFunction() {
      document.getElementById("demo").innerHTML = "I was double-clicked!";
    }

Jquery

Code 
    $("p").dblclick(function(){

    });

Moves the mouse

Explanation: When the user moves the mouse cursor

Javascript Vanilla

Code 
    button.onclick = function(event) {
      if (event.altKey && event.shiftKey) {
        alert('Hooray!');
      }
    };

    On HTML script: onmousemove="this.value=event.clientX+':'+event.clientY"

Jquery

Code 
    $( "div" ).mousemove(function( event ) {
      var pageCoords = "( " + event.pageX + ", " + event.pageY + " )";
      var clientCoords = "( " + event.clientX + ", " + event.clientY + " )";
      $( "span" ).first().text( "( event.pageX, event.pageY ) : " + pageCoords );
      $( "span" ).last().text( "( event.clientX, event.clientY ) : " + clientCoords );
    });

    $( "#cart" ).on( "mouseenter mouseleave", function( event ) {
      $( this ).toggleClass( "active" );
    });

Image fails to load

Explanation: When an image fails to load, (if you write an incorrect image url the loading of the image will fail)

Javascript Vanilla

Code 
    function testImage(url) {
      var tester = new Image();
      tester.addEventListener('load', imageFound);
      tester.addEventListener('error', imageNotFound);
      tester.src = url;
    }
    function imageFound() {
        alert('That image is found and loaded');
    }
    function imageNotFound() {
        alert('That image was not found.');
    }
    testImage("http://foo.com/bar.jpg");

Jquery

Code 
    $("img").on("error", function(){
      $("img").replaceWith("<p>Error loading image!</p>");
    });

Changes the option of a select element

Explanation: When the user changes the option of a select element

Javascript Vanilla

Code 
    function select(selectId, optionValToSelect){
        var selectElement = document.getElementById(selectId);
        var selectOptions = selectElement.options;
        for (var opt, j = 0; opt = selectOptions[j]; j++) {
            if (opt.value == optionValToSelect) {
                selectElement.selectedIndex = j;
                break;
            }
        }
    }

Jquery

Code 
    $( ".target" ).change(function() {
      alert( "Handler for .change() called." );
    });

    $( "#other" ).click(function() {
      $( ".target" ).change();
    });

Checked or unchecked

Explanation: When a checkbox is checked or unchecked

Javascript Vanilla

Code 
      function check() {
      document.getElementById("myCheck").checked = true;
    }

    function uncheck() {
      document.getElementById("myCheck").checked = false;
    }

Jquery

Code 
    $("#checkbox_activar").click(function() {
      $("#checkbox").attr('checked', true);
    });

    $("#checkbox_desactivar").click(function() {
      $("#checkbox").attr('checked', false);
    });

    $( "input" ).on( "click", function() {
      $( "#log" ).html( $( "input:checked" ).val() + " is checked!" );
    });

Show a list when clicked

Explanation: When a ul list item is clicked, show the item that was clicked

Javascript Vanilla

Code 
    function fAbreGuias() {
      var vGuias = document.getElementById('guias');
      if (vGuias.style.display == "block") {
        vGuias.style.display = "none";
      } else {
        vGuias.style.display = "block";
      }
    }

Jquery

Code 
    $( "button" ).click(function() {
      $( "p" ).show( "slow" );
    });

    $( "#showr" ).click(function() {
      $( "div" ).first().show( "fast", function showNext() {
        $( this ).next( "div" ).show( "fast", showNext );
      });
    });

    $( "#hidr" ).click(function() {
      $( "div" ).hide( 1000 );
    });
